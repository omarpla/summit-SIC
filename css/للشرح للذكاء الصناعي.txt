class InteractiveServices {
  constructor() {
    this.grid = document.getElementById('servicesGrid');
    this.cards = Array.from(document.querySelectorAll('.service-card'));
    this.toggles = Array.from(document.querySelectorAll('.service-toggle, .close-details'));
    this.activeCard = null;
    
    this.init();
  }
  
  init() {
    // Initialize event listeners
    this.bindEvents();
    
    // Set initial state
    this.updateGridState();
    
    // Dispatch ready event
    this.dispatchEvent('services:ready');
  }
  
  bindEvents() {
    // Card click events
    this.cards.forEach(card => {
      card.addEventListener('click', (e) => {
        if (!e.target.closest('.service-toggle, .close-details')) {
          this.toggleCard(card);
        }
      });
    });
    
    // Toggle button events
    this.toggles.forEach(toggle => {
      toggle.addEventListener('click', (e) => {
        e.stopPropagation();
        const card = toggle.closest('.service-card');
        
        if (toggle.dataset.action === 'expand') {
          this.expandCard(card);
        } else {
          this.collapseCard(card);
        }
      });
    });
    
    // Expand/Collapse all buttons
    const expandAllBtn = document.getElementById('expandAll');
    const collapseAllBtn = document.getElementById('collapseAll');
    
    if (expandAllBtn) {
      expandAllBtn.addEventListener('click', () => this.expandAll());
    }
    
    if (collapseAllBtn) {
      collapseAllBtn.addEventListener('click', () => this.collapseAll());
    }
    
    // Close on escape key
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Escape' && this.activeCard) {
        this.collapseCard(this.activeCard);
      }
    });
  }
  
  toggleCard(card) {
    if (card.classList.contains('active')) {
      this.collapseCard(card);
    } else {
      this.expandCard(card);
    }
  }
  
  expandCard(card) {
    // Collapse current active card if exists
    if (this.activeCard && this.activeCard !== card) {
      this.collapseCard(this.activeCard);
    }
    
    // Set states
    this.activeCard = card;
    card.classList.add('active');
    card.classList.remove('inactive');
    
    // Update other cards
    this.cards.forEach(c => {
      if (c !== card) {
        c.classList.add('inactive');
        c.classList.remove('active');
      }
    });
    
    // Update grid state
    this.updateGridState();
    
    // Dispatch event
    this.dispatchEvent('service:expand', { card });
    
    // Scroll card into view (smoothly)
    setTimeout(() => {
      card.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
    }, 300);
  }
  
  collapseCard(card) {
    card.classList.remove('active');
    
    // If collapsing the active card
    if (this.activeCard === card) {
      this.activeCard = null;
    }
    
    // Reset other cards
    this.cards.forEach(c => {
      if (c !== card) {
        c.classList.remove('inactive');
      }
    });
    
    // Update grid state
    this.updateGridState();
    
    // Dispatch event
    this.dispatchEvent('service:collapse', { card });
  }
  
  expandAll() {
    this.cards.forEach(card => {
      card.classList.add('active');
      card.classList.remove('inactive');
    });
    
    this.grid.classList.add('has-active');
    this.dispatchEvent('services:expandAll');
  }
  
  collapseAll() {
    this.cards.forEach(card => {
      card.classList.remove('active', 'inactive');
    });
    
    this.activeCard = null;
    this.updateGridState();
    this.dispatchEvent('services:collapseAll');
  }
  
  updateGridState() {
    if (this.activeCard) {
      this.grid.classList.add('has-active');
    } else {
      this.grid.classList.remove('has-active');
    }
  }
  
  dispatchEvent(eventName, detail = {}) {
    const event = new CustomEvent(eventName, {
      detail: { ...detail, system: this }
    });
    document.dispatchEvent(event);
  }
  
  // Public API
  getActiveService() {
    return this.activeCard;
  }
  
  getServicesCount() {
    return this.cards.length;
  }
}

class MultiMarkerMap {
  constructor() {
    this.map = document.getElementById('interactiveMap');
    this.markers = Array.from(document.querySelectorAll('.map-marker'));
    this.controls = {
      zoomIn: document.getElementById('zoomIn'),
      zoomOut: document.getElementById('zoomOut'),
      reset: document.getElementById('resetMap'),
      toggleType: document.getElementById('toggleMapType')
    };
    
    this.activeMarker = null;
    this.mapType = 'roadmap'; // roadmap, satellite, hybrid, terrain
    this.zoomLevel = 14;
    
    this.init();
  }
  
  init() {
    // Initialize event listeners
    this.bindEvents();
    
    // Set initial marker states
    this.updateMarkers();
    
    // Dispatch ready event
    this.dispatchEvent('map:ready');
  }
  
  bindEvents() {
    // Marker click events
    this.markers.forEach(marker => {
      marker.addEventListener('click', (e) => {
        e.stopPropagation();
        this.activateMarker(marker);
      });
      
      marker.addEventListener('mouseenter', () => {
        marker.classList.add('hover');
      });
      
      marker.addEventListener('mouseleave', () => {
        marker.classList.remove('hover');
      });
    });
    
    // Map control events
    if (this.controls.zoomIn) {
      this.controls.zoomIn.addEventListener('click', () => this.zoomIn());
    }
    
    if (this.controls.zoomOut) {
      this.controls.zoomOut.addEventListener('click', () => this.zoomOut());
    }
    
    if (this.controls.reset) {
      this.controls.reset.addEventListener('click', () => this.resetMap());
    }
    
    if (this.controls.toggleType) {
      this.controls.toggleType.addEventListener('click', () => this.toggleMapType());
    }
    
    // Directions and address copy buttons
    const directionsBtn = document.getElementById('getDirections');
    const copyAddressBtn = document.getElementById('copyAddress');
    
    if (directionsBtn) {
      directionsBtn.addEventListener('click', () => this.getDirections());
    }
    
    if (copyAddressBtn) {
      copyAddressBtn.addEventListener('click', () => this.copyAddress());
    }
    
    // Close active marker on outside click
    document.addEventListener('click', (e) => {
      if (this.activeMarker && !this.activeMarker.contains(e.target)) {
        this.deactivateMarker(this.activeMarker);
      }
    });
    
    // Close on escape key
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Escape' && this.activeMarker) {
        this.deactivateMarker(this.activeMarker);
      }
    });
  }
  
  activateMarker(marker) {
    // Deactivate current active marker
    if (this.activeMarker && this.activeMarker !== marker) {
      this.deactivateMarker(this.activeMarker);
    }
    
    // Toggle if clicking the same marker
    if (this.activeMarker === marker) {
      this.deactivateMarker(marker);
      return;
    }
    
    // Activate new marker
    marker.classList.add('active');
    this.activeMarker = marker;
    
    // Center map on marker
    this.centerOnMarker(marker);
    
    // Dispatch event
    this.dispatchEvent('marker:activate', { marker });
  }
  
  deactivateMarker(marker) {
    marker.classList.remove('active');
    
    if (this.activeMarker === marker) {
      this.activeMarker = null;
    }
    
    this.dispatchEvent('marker:deactivate', { marker });
  }
  
  centerOnMarker(marker) {
    const lat = marker.dataset.lat || '29.2985';
    const lng = marker.dataset.lng || '47.6855';
    
    // Update map iframe src to center on marker
    this.updateMapView(lat, lng, this.zoomLevel);

function optimizeServicesMapPerformance() {

  const serviceImages = document.querySelectorAll('.media-image');
  
  if ('IntersectionObserver' in window) {

      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const img = entry.target;
          if (img.dataset.src) {
            img.src = img.dataset.src;
            img.removeAttribute('data-src');
          }
          observer.unobserve(img);
        }
      });
    });
    
    serviceImages.forEach(img => observer.observe(img));
  }
  
  // Use will-change for performance
  const interactiveElements = document.querySelectorAll('.service-card, .map-marker, .map-btn');
  interactiveElements.forEach(el => {
    el.addEventListener('mouseenter', () => {
      el.style.willChange = 'transform, opacity';
    });
    
    el.addEventListener('mouseleave', () => {
      el.style.willChange = 'auto';
    });
  });
}

// 1. Back to Top Logic
const backToTopBtn = document.getElementById('backToTop');
window.addEventListener('scroll', () => {
  if (window.scrollY > 300) {
    backToTopBtn.classList.add('visible');
  } else {
    backToTopBtn.classList.remove('visible');
  }
});

backToTopBtn.addEventListener('click', () => {
  window.scrollTo({ top: 0, behavior: 'smooth' });
});

    icon.classList.replace('fa-times', 'fa-share-alt');
